void storeUserInRedis(struct User *user, long id) {
    redisReply *reply = (redisReply *)redisCommand(NULL, "HMSET user:%ld name %s age %d", id, user->name, user->age);
    freeReplyObject(reply);
}

void getUserFromRedis(long id, struct User *user) {
    redisReply *reply = (redisReply *)redisCommand(NULL, "HGETALL user:%ld", id);
    if (reply->type == REDIS_REPLY_ARRAY && reply->elements >= 4) {
        user->name = strdup(reply->element[1]->str);
        user->age = atoi(reply->element[3]->str);
    }
    freeReplyObject(reply);
}

void printUser(const struct User *user) {
    printf("Name: %s, Age: %d\n", user->name, user->age);
