工厂设计模式

#include <iostream>
#include <memory>

using namespace std; // 引入 std 命名空间

// 抽象产品类
class Product {
public:
    virtual void display() = 0;
    virtual ~Product() {}
};

// 具体产品类 A
class ConcreteProductA : public Product {
public:
    void display() override {
        cout << "Concrete Product A" << endl;
    }
};

// 具体产品类 B
class ConcreteProductB : public Product {
public:
    void display() override {
        cout << "Concrete Product B" << endl;
    }
};

// 工厂类
class Factory {
public:
    // 静态方法，根据参数创建具体产品对象
    static unique_ptr<Product> createProduct(char type) {
        switch (type) {
            case 'A':
                return make_unique<ConcreteProductA>();
            case 'B':
                return make_unique<ConcreteProductB>();
            default:
                return nullptr;
        }
    }
};

int main() {
    // 使用工厂创建具体产品 A
    unique_ptr<Product> productA = Factory::createProduct('A');
    if (productA) {
        productA->display();
    }

    // 使用工厂创建具体产品 B
    unique_ptr<Product> productB = Factory::createProduct('B');
    if (productB) {
        productB->display();
    }

    // 使用工厂创建不存在的产品
    unique_ptr<Product> productC = Factory::createProduct('C');
    if (!productC) {
        cout << "Invalid product type requested." << endl;
    }

    return 0;
}
